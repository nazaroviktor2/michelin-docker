# michelin-docker
# How install
$ git clone https://github.com/nazaroviktor2/michelin-docker.git

# How run
- add you data to .env

$ sudo apt-get update

$ sudo apt-get install docker-ce docker-ce-cli containerd.io docker-compose-plugin

$ chmod +x init.sh

$ chmod +x start.sh

$ ./init.sh

$ ./start.sh
